-- 03-b-4003

```
#!/bin/bash

CNT=$(cat "$1" | sed -E 's/(.)/\1\n/g' | sort | grep '[^\n]' | uniq -c | sort -k 1 -n -r | head -n 1 | awk '{print $1;}')

cat "$1" | sed -E 's/(.)/\1\n/g' | sort | grep '[^\n]' | uniq -c | sort -k 1 -n -r | awk '{printf("%s %s\n", $1, $2);}' 
| grep "^${CNT} " | awk '{print $2;}' | sort
```
